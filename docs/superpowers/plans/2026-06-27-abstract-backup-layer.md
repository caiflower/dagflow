# Abstract Backup Layer 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 backup 逻辑抽象为 `BackupManager` 接口，MySQL 走冷表归档+清理，Redis 直接删除超龄 key；新增 `TaskQueryService` 支持主表+archive 联合查询；改造 `ExecutionService` 使用新查询接口。

**Architecture:** 在 `backend/taskx/` 包内新建 `backup_manager.go` 和 `task_query_service.go`，定义 `BackupManager` / `TaskQueryService` 接口及 sql/redis 双实现。通过 `InitTaskDispatcher` 中的 `StorageBackend` 判断将对应实现注入 bean 容器。冷表命名统一为 `task_archive` / `subtask_archive` / `task_edge_archive`。

**Tech Stack:** Go 1.24, bun ORM (MySQL), go-redis/v9, caiflower common-tools (bean/IoC, db/v1, redis/v2)

## 全局约束

- 不新增 gRPC 接口
- 不改动非 backup 相关的 DAO 接口方法
- 不改变 Redis key 命名规则
- 不引入新的外部依赖
- 模型字段与对应主表模型完全一致（仅 `bun.BaseModel` 嵌入）
- `backupTask()` 当前已被注释掉，本次只改方法体，不改变调用方式
- DDL（表重命名+新建 `task_edge_archive`）不属于本次代码变更范围

// GetByIDs 批量按 ID 查询归档任务
func (d *taskBakDAO) GetByIDs(ctx context.Context, ids []string) ([]model.TaskBak, error) {
	if len(ids) == 0 {
		return nil, nil
	}
	var tasks []model.TaskBak
	err := d.client.GetDB().NewSelect().Model(&tasks).ModelTableExpr(d.tableName).ColumnExpr("*").Where("status>0").Where("id IN (?)", bun.In(ids)).Scan(ctx)
	if err != nil {
		if d.client.ParseErr(err) == nil {
			return nil, nil
		}
		return nil, err
	}
	return tasks, nil
}

// BatchInsert 批量插入归档任务
func (d *taskBakDAO) BatchInsert(ctx context.Context, data []model.TaskBak) (int64, error) {
	if len(data) == 0 {
		return 0, nil
	}
	pageNumber := 1
	batch := 50
	count := int64(0)
	for {
		canSplit, start, end := dbv1.SplitIndex(pageNumber, batch, len(data))
		if !canSplit {
			break
		}
		batchList := data[start:end]
		cnt, err := d.client.GetRowsAffected(d.db(ctx).NewInsert().Model(&batchList).ModelTableExpr(d.tableName).Exec(ctx))
		if err != nil {
			return count, err
		}
		count += cnt
		pageNumber++
	}
	return count, nil
}

package taskx

import (
	"context"
	"time"

	dbv1 "github.com/caiflower/common-tools/db/v1"
	"github.com/caiflower/common-tools/pkg/basic"
	"github.com/caiflower/common-tools/pkg/logger"
	"github.com/caiflower/dagflow/taskx/dao"
	"github.com/caiflower/dagflow/taskx/dao/model"
	"github.com/uptrace/bun"
)

// BackupConfig 冷数据备份配置
type BackupConfig struct {
	Age         time.Duration // 超龄阈值：MySQL 默认 168h，Redis 默认 24h
	BatchSize   int           // 每批处理数量，默认 100
	FinalStates []string      // 终态列表，默认 ["failed", "succeeded"]
}

// BackupManager 冷数据备份管理器接口
type BackupManager interface {
	BackupTasks(ctx context.Context, cfg BackupConfig) (int, error)
}

// sqlBackupManager MySQL 备份实现：冷数据迁移到 archive 表并清理原表
type sqlBackupManager struct {
	DBClient      dbv1.DB           `autowired:""`
	TaskDao       dao.TaskDAO       `autowired:""`
	TaskBakDao    dao.TaskBakDAO    `autowired:""`
	SubtaskDao    dao.SubtaskDAO    `autowired:""`
	SubtaskBakDao dao.SubtaskBakDAO `autowired:""`
	TaskEdgeDao   dao.TaskEdgeDAO   `autowired:""`
	TaskEdgeBakDao dao.TaskEdgeBakDAO `autowired:""`
}

func (m *sqlBackupManager) BackupTasks(ctx context.Context, cfg BackupConfig) (int, error) {
	// 1. 查询超龄任务
	deadline := time.Now().Add(-cfg.Age)
	tasks, err := m.TaskDao.GetTodoTask(ctx, cfg.FinalStates, basic.Time(deadline))
	if err != nil {
		logger.Error("[sqlBackupManager] query tasks for backup failed. err: %v", err)
		return 0, err
	}
	if len(tasks) == 0 {
		return 0, nil
	}

	// 2. 提取 taskIDs
	taskIDs := make([]string, len(tasks))
	for i, t := range tasks {
		taskIDs[i] = t.ID
	}

	// 3. 转换 Task → TaskBak
	taskBaks := make([]model.TaskBak, len(tasks))
	for i, t := range tasks {
		taskBaks[i] = taskToTaskBak(&t)
	}

	// 4. 批量事务：插入 archive 表 + 删除原表
	tx := dbv1.NewBatchTx(m.DBClient.GetDB())

	tx.Add(func(tx *bun.Tx) error {
		_, err := m.TaskBakDao.BatchInsert(ctx, taskBaks)
		if err != nil {
			logger.Error("[sqlBackupManager] insert task archive failed. err: %v", err)
			return err
		}
		return nil
	})

	// 5. 查询 subtask + edge 并归档
	for _, taskID := range taskIDs {
		tid := taskID
		tx.Add(func(tx *bun.Tx) error {
			subtasks, err := m.SubtaskDao.GetByTaskID(ctx, tid)
			if err != nil {
				return err
			}
			subtaskBaks := make([]model.SubtaskBak, len(subtasks))
			for i, s := range subtasks {
				subtaskBaks[i] = subtaskToSubtaskBak(&s)
			}
			if len(subtaskBaks) > 0 {
				if _, err := m.SubtaskBakDao.BatchInsert(ctx, subtaskBaks); err != nil {
					logger.Error("[sqlBackupManager] insert subtask archive failed for task %s. err: %v", tid, err)
					return err
				}
			}
			return nil
		})

		tx.Add(func(tx *bun.Tx) error {
			edges, err := m.TaskEdgeDao.GetByTaskID(ctx, tid)
			if err != nil {
				return err
			}
			edgeBaks := make([]model.TaskEdgeBak, len(edges))
			for i, e := range edges {
				edgeBaks[i] = taskEdgeToTaskEdgeBak(&e)
			}
			if len(edgeBaks) > 0 {
				if _, err := m.TaskEdgeBakDao.BatchInsert(ctx, edgeBaks); err != nil {
					logger.Error("[sqlBackupManager] insert edge archive failed for task %s. err: %v", tid, err)
					return err
				}
			}
			return nil
		})
	}

	// 6. 删除原表数据（逐 task 处理）
	for _, taskID := range taskIDs {
		tid := taskID
		// 删除 subtask
		tx.Add(func(tx *bun.Tx) error {
			result, err := tx.NewDelete().Table("subtask").Where("task_id = ?", tid).Exec(ctx)
			if err != nil {
				logger.Error("[sqlBackupManager] delete subtasks failed for task %s. err: %v", tid, err)
				return err
			}
			rows, _ := result.RowsAffected()
			logger.Info("[sqlBackupManager] deleted %d subtasks for task %s", rows, tid)
			return nil
		})
		// 删除 task_edge
		tx.Add(func(tx *bun.Tx) error {
			result, err := tx.NewDelete().Table("task_edge").Where("task_id = ?", tid).Exec(ctx)
			if err != nil {
				logger.Error("[sqlBackupManager] delete task_edges failed for task %s. err: %v", tid, err)
				return err
			}
			rows, _ := result.RowsAffected()
			logger.Trace("[sqlBackupManager] deleted %d task_edges for task %s", rows, tid)
			return nil
		})
		// 删除 task
		tx.Add(func(tx *bun.Tx) error {
			result, err := tx.NewDelete().Table("task").Where("id = ?", tid).Exec(ctx)
			if err != nil {
				logger.Error("[sqlBackupManager] delete task failed for task %s. err: %v", tid, err)
				return err
			}
			rows, _ := result.RowsAffected()
			logger.Info("[sqlBackupManager] deleted task %s (%d rows)", tid, rows)
			return nil
		})
	}

	if err := tx.Submit(); err != nil {
		logger.Error("[sqlBackupManager] tx submit failed. err: %v", err)
		return 0, err
	}

	logger.Info("[sqlBackupManager] successfully archived and deleted %d tasks: %v", len(taskIDs), taskIDs)
	return len(taskIDs), nil
}

// redisBackupManager Redis 备份实现：直接删除超龄 key
type redisBackupManager struct {
	Client        v2.RedisClient   `autowired:""`
	TaskDao       dao.TaskDAO      `autowired:""`
	SubtaskDao    dao.SubtaskDAO   `autowired:""`
	TaskEdgeDao   dao.TaskEdgeDAO  `autowired:""`
}

func (m *redisBackupManager) BackupTasks(ctx context.Context, cfg BackupConfig) (int, error) {
	// 1. 查询超龄任务
	deadline := time.Now().Add(-cfg.Age)
	tasks, err := m.TaskDao.GetTodoTask(ctx, cfg.FinalStates, basic.Time(deadline))
	if err != nil {
		logger.Error("[redisBackupManager] query tasks for backup failed. err: %v", err)
		return 0, err
	}
	if len(tasks) == 0 {
		return 0, nil
	}

	c := m.Client.Cmd()
	deletedCount := 0

	for _, task := range tasks {
		// 2. 删除 subtask 数据
		subtaskIDs, err := c.SMembers(ctx, fmt.Sprintf("dagflow:task:{%s}:subtasks", task.ID)).Result()
		if err == nil {
			for _, sid := range subtaskIDs {
				c.Del(ctx, fmt.Sprintf("dagflow:subtask:{%s}", sid))
			}
		}
		c.Del(ctx, fmt.Sprintf("dagflow:task:{%s}:subtasks", task.ID))

		// 3. 删除 task edge 数据
		edgeIDs, err := c.SMembers(ctx, fmt.Sprintf("dagflow:task:{%s}:edges", task.ID)).Result()
		if err == nil {
			for _, eid := range edgeIDs {
				c.Del(ctx, fmt.Sprintf("dagflow:edge:{%s}", eid))
			}
		}
		c.Del(ctx, fmt.Sprintf("dagflow:task:{%s}:edges", task.ID))

		// 4. 删除 task 自身
		c.Del(ctx, fmt.Sprintf("dagflow:task:{%s}", task.ID))
		// 5. 从 todo set 中移除
		c.ZRem(ctx, "dagflow:todo", task.ID)

		deletedCount++
	}

	logger.Info("[redisBackupManager] deleted %d expired tasks from Redis", deletedCount)
	return deletedCount, nil
}

package taskx

import (
	"context"
	"fmt"

	"github.com/caiflower/common-tools/pkg/logger"
	"github.com/caiflower/dagflow/taskx/dao"
	"github.com/caiflower/dagflow/taskx/dao/model"
)

// TaskDetail 聚合一个 Task 及其 Subtask/Edge 的完整信息
type TaskDetail struct {
	Task     model.Task
	Subtasks []model.Subtask
	Edges    []model.TaskEdge
}

// TaskQueryService 任务查询服务接口：支持主表+archive 联合查询
type TaskQueryService interface {
	GetTasks(ctx context.Context, taskIDs []string) ([]TaskDetail, error)
}

// sqlTaskQueryService MySQL 实现：主表 + archive 表补查
type sqlTaskQueryService struct {
	TaskDao        dao.TaskDAO        `autowired:""`
	TaskBakDao     dao.TaskBakDAO     `autowired:""`
	SubtaskDao     dao.SubtaskDAO     `autowired:""`
	SubtaskBakDao  dao.SubtaskBakDAO  `autowired:""`
	TaskEdgeDao    dao.TaskEdgeDAO    `autowired:""`
	TaskEdgeBakDao dao.TaskEdgeBakDAO `autowired:""`
}

func (s *sqlTaskQueryService) GetTasks(ctx context.Context, taskIDs []string) ([]TaskDetail, error) {
	if len(taskIDs) == 0 {
		return nil, nil
	}

	// 1. 从主表查询
	tasks, err := s.TaskDao.GetByIDs(ctx, taskIDs)
	if err != nil {
		return nil, fmt.Errorf("sqlTaskQueryService GetByIDs: %w", err)
	}

	hitMap := make(map[string]bool, len(tasks))
	for i := range tasks {
		hitMap[tasks[i].ID] = true
	}

	// 2. 计算缺失的 taskIDs，从 archive 表补查
	var missingIDs []string
	for _, id := range taskIDs {
		if !hitMap[id] {
			missingIDs = append(missingIDs, id)
		}
	}

	var archivedTasks []model.TaskBak
	if len(missingIDs) > 0 {
		archivedTasks, err = s.TaskBakDao.GetByIDs(ctx, missingIDs)
		if err != nil {
			logger.Error("[sqlTaskQueryService] query archive tasks failed. err: %v", err)
		}
	}

	// 3. 组装结果
	result := make([]TaskDetail, 0, len(taskIDs))
	seen := make(map[string]bool, len(taskIDs))

	for i := range tasks {
		t := &tasks[i]
		if seen[t.ID] {
			continue
		}
		seen[t.ID] = true

		detail := TaskDetail{Task: *t}

		// 查询 subtask + edge
		subtasks, _ := s.SubtaskDao.GetByTaskID(ctx, t.ID)
		detail.Subtasks = subtasks

		edges, _ := s.TaskEdgeDao.GetByTaskID(ctx, t.ID)
		detail.Edges = edges

		result = append(result, detail)
	}

	for i := range archivedTasks {
		bak := &archivedTasks[i]
		if seen[bak.ID] {
			continue
		}
		seen[bak.ID] = true

		taskModel := taskBakToTask(bak)
		detail := TaskDetail{Task: *taskModel}

		// 从 archive 表查 subtask + edge
		subtaskBaks, _ := s.SubtaskBakDao.GetByTaskID(ctx, bak.ID)
		subtasks := make([]model.Subtask, len(subtaskBaks))
		for j, sb := range subtaskBaks {
			subtasks[j] = *subtaskBakToSubtask(&sb)
		}
		detail.Subtasks = subtasks

		edgeBaks, _ := s.TaskEdgeBakDao.GetByTaskID(ctx, bak.ID)
		edges := make([]model.TaskEdge, len(edgeBaks))
		for j, eb := range edgeBaks {
			edges[j] = *taskEdgeBakToTaskEdge(&eb)
		}
		detail.Edges = edges

		result = append(result, detail)
	}

	return result, nil
}

	// 1. 从执行记录表查 task_id
	record, err := s.ExecutionRecordDAO.GetByID(ctx, execID)
